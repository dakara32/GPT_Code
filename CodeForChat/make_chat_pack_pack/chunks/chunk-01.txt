## <tools/make_chat_pack.py>
```text
L1 #!/usr/bin/env python3
L2 # -*- coding: utf-8 -*-
L3 """
L4 make_chat_pack.py
L5 - FILES_GROUPS（2パック）を読み込み
L6 - 各ファイルに L1.. の行番号を付与し連結
L7 - CodeForChat/<pack>/chat-pack.txt と chunks/ を生成
L8 - CodeForChat/INDEX.txt（全パック目次）も出力
L9 """
L10
L11 import os, sys, textwrap, hashlib
L12 import datetime
L13
L14 OWNER  = "dakara32"
L15 REPO   = "GPT_Code"
L16 BRANCH = "main"
L17
L18 CHUNK_SIZE = 12000  # iPhoneコピペ配慮で ~9k–14k の範囲で可
L19
L20 FILES_GROUPS = {
L21     "factor_pack": [
L22         "config.py",
L23         "factor.py",
L24         "scorer.py",
L25         ".github/workflows/weekly-report.yml",
L26         "documents/README.md",
L27         "documents/factor_design.md",
L28     ],
L29     "drift_pack": [
L30         "config.py",
L31         "drift.py",
L32         ".github/workflows/daily-report.yml",
L33         "documents/README.md",
L34         "documents/drift_design.md",
L35     ],
L36     "yf-health-probe_pack": [
L37         "tools/yf_health_probe.py",
L38         ".github/workflows/yf-health-probe.yml",
L39     ],  
L40     "make_chat_pack_pack": [
L41         "tools/make_chat_pack.py",
L42         ".github/workflows/prepare-chat-pack.yml",
L43     ],      
L44 }
L45
L46 def add_line_numbers(body: str) -> str:
L47     out = []
L48     for i, line in enumerate(body.splitlines(), start=1):
L49         out.append(f"L{i}{'' if line=='' else ' '}{line}")
L50     return "\n".join(out) if out else "L1"
L51
L52 def read_file(path: str) -> str:
L53     with open(path, "r", encoding="utf-8", errors="replace") as f:
L54         return f.read()
L55
L56 def header_block(file_list) -> str:
L57     JST = datetime.timezone(datetime.timedelta(hours=9))
L58     now = datetime.datetime.now(JST).strftime("%Y-%m-%d %H:%M:%S")
L59     return textwrap.dedent(f"""\
L60     # === Chat Paste Pack ===
L61     # Repo: {OWNER}/{REPO} @ {BRANCH}
L62     # Files: {', '.join(file_list)}
L63     # 作成日時: {now} (JST)
L64     # 使い方: 下のチャンクを順に貼ればこのチャットで全体把握できます。
L65     # 注記: 各ファイルは個別に L1.. で行番号付与。
L66     ---
L67     """)
L68
L69 def make_bundle(file_list) -> str:
L70     parts = [header_block(file_list)]
L71     for fp in file_list:
L72         if not os.path.exists(fp):
L73             parts.append(f"## <{fp}> (NOT FOUND)\n```text\nL1\n```\n")
L74             continue
L75         body = read_file(fp)
L76         parts.append(f"## <{fp}>\n```text\n{add_line_numbers(body)}\n```\n")
L77     return "\n".join(parts)
L78
L79 def chunkify(s: str, n: int):
L80     return [s[i:i+n] for i in range(0, len(s), n)]
L81
L82 def main():
L83     root_out = "CodeForChat"
L84     os.makedirs(root_out, exist_ok=True)
L85
L86     global_index = ["# Chat Paste Pack Index", ""]
L87
L88     for pack_name, file_list in FILES_GROUPS.items():
L89         bundle = make_bundle(file_list)
L90         sha = hashlib.sha1(bundle.encode("utf-8")).hexdigest()[:12]
L91
L92         pack_dir  = os.path.join(root_out, pack_name)
L93         chunk_dir = os.path.join(pack_dir, "chunks")
L94         os.makedirs(chunk_dir, exist_ok=True)
L95
L96         # full
L97         with open(os.path.join(pack_dir, "chat-pack.txt"), "w", encoding="utf-8") as w:
L98             w.write(bundle)
L99
L100         # chunks
L101         chunks = chunkify(bundle, CHUNK_SIZE)
L102         local_index = [f"# {pack_name} (SHA:{sha}) / {len(chunks)} chunks", ""]
L103         for idx, raw in enumerate(chunks, start=1):
L104             with open(os.path.join(chunk_dir, f"chunk-{idx:02d}.txt"), "w", encoding="utf-8") as w:
L105                 w.write(f"```text\n{raw}\n```")
L106             local_index.append(f"- chunks/chunk-{idx:02d}.txt")
L107
L108         with open(os.path.join(pack_dir, "INDEX.txt"), "w", encoding="utf-8") as w:
L109             w.write("\n".join(local_index))
L110
L111         global_index += [f"## {pack_name}",
L112                          f"- {pack_name}/chat-pack.txt",
L113                          f"- {pack_name}/INDEX.txt", ""]
L114
L115     with open(os.path.join(root_out, "INDEX.txt"), "w", encoding="utf-8") as w:
L116         w.write("\n".join(global_index))
L117
L118 if __name__ == "__main__":
L119     sys.exit(main())
```
