L281 def _sticky_keep_current(agg: pd.Series, pick: list[str], incumbents: list[str],
L282                          n_target: int, delta_z: float, keep_buffer: int) -> list[str]:
L283     import pandas as pd, numpy as np
L284     sel = list(pick)
L285     if not sel: return sel
L286     ranked_sel = agg.reindex(sel).sort_values(ascending=False)
L287     kth = ranked_sel.iloc[min(len(sel), n_target)-1]
L288     sigma = float(agg.std()) if pd.notna(agg.std()) else 0.0
L289     thresh = kth - delta_z * sigma
L290     ranked_all = agg.sort_values(ascending=False)
L291     cand = [t for t in incumbents if (t not in sel) and (t in agg.index)]
L292     for t in cand:
L293         within_score = (pd.notna(agg[t]) and agg[t] >= thresh)
L294         within_rank  = (t in ranked_all.index) and (ranked_all.index.get_loc(t) < n_target + keep_buffer)
L295         if within_score or within_rank:
L296             non_inc = [x for x in sel if x not in incumbents]
L297             if not non_inc: break
L298             weakest = min(non_inc, key=lambda x: agg.get(x, -np.inf))
L299             if weakest in sel and agg.get(t, -np.inf) >= agg.get(weakest, -np.inf):
L300                 sel.remove(weakest); sel.append(t)
L301     if len(sel) > n_target:
L302         sel = sorted(sel, key=lambda x: agg.get(x, -1e9), reverse=True)[:n_target]
L303     return sel
