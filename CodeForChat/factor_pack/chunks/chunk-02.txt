L304 def _sticky_keep_current(agg: pd.Series, pick: list[str], incumbents: list[str],
L305                          n_target: int, delta_z: float, keep_buffer: int) -> list[str]:
L306     import pandas as pd, numpy as np
L307     sel = list(pick)
L308     if not sel: return sel
L309     ranked_sel = agg.reindex(sel).sort_values(ascending=False)
L310     kth = ranked_sel.iloc[min(len(sel), n_target)-1]
L311     sigma = float(agg.std()) if pd.notna(agg.std()) else 0.0
L312     thresh = kth - delta_z * sigma
L313     ranked_all = agg.sort_values(ascending=False)
L314     cand = [t for t in incumbents if (t not in sel) and (t in agg.index)]
L315     for t in cand:
L316         within_score = (pd.notna(agg[t]) and agg[t] >= thresh)
L317         within_rank  = (t in ranked_all.index) and (ranked_all.index.get_loc(t) < n_target + keep_buffer)
L318         if within_score or within_rank:
L319             non_inc = [x for x in sel if x not in incumbents]
L320             if not non_inc: break
L321             weakest = min(non_inc, key=lambda x: agg.get(x, -np.inf))
L322             if weakest in sel and agg.get(t, -np.inf) >= agg.get(weakest, -np.inf):
L323                 sel.remove(weakest); sel.append(t)
L324     if len(sel) > n_target:
L325         sel = sorted(sel, key=lambda x: agg.get(x, -1e9), reverse=True)[:n_target]
L326     return sel
