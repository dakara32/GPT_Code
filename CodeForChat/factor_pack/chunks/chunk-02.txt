L306 def _sticky_keep_current(agg: pd.Series, pick: list[str], incumbents: list[str],
L307                          n_target: int, delta_z: float, keep_buffer: int) -> list[str]:
L308     import pandas as pd, numpy as np
L309     sel = list(pick)
L310     if not sel: return sel
L311     ranked_sel = agg.reindex(sel).sort_values(ascending=False)
L312     kth = ranked_sel.iloc[min(len(sel), n_target)-1]
L313     sigma = float(agg.std()) if pd.notna(agg.std()) else 0.0
L314     thresh = kth - delta_z * sigma
L315     ranked_all = agg.sort_values(ascending=False)
L316     cand = [t for t in incumbents if (t not in sel) and (t in agg.index)]
L317     for t in cand:
L318         within_score = (pd.notna(agg[t]) and agg[t] >= thresh)
L319         within_rank  = (t in ranked_all.index) and (ranked_all.index.get_loc(t) < n_target + keep_buffer)
L320         if within_score or within_rank:
L321             non_inc = [x for x in sel if x not in incumbents]
L322             if not non_inc: break
L323             weakest = min(non_inc, key=lambda x: agg.get(x, -np.inf))
L324             if weakest in sel and agg.get(t, -np.inf) >= agg.get(weakest, -np.inf):
L325                 sel.remove(weakest); sel.append(t)
L326     if len(sel) > n_target:
L327         sel = sorted(sel, key=lambda x: agg.get(x, -1e9), reverse=True)[:n_target]
L328     return sel
