L324 def _sticky_keep_current(agg: pd.Series, pick: list[str], incumbents: list[str],
L325                          n_target: int, delta_z: float, keep_buffer: int) -> list[str]:
L326     import pandas as pd, numpy as np
L327     sel = list(pick)
L328     if not sel: return sel
L329     ranked_sel = agg.reindex(sel).sort_values(ascending=False)
L330     kth = ranked_sel.iloc[min(len(sel), n_target)-1]
L331     sigma = float(agg.std()) if pd.notna(agg.std()) else 0.0
L332     thresh = kth - delta_z * sigma
L333     ranked_all = agg.sort_values(ascending=False)
L334     cand = [t for t in incumbents if (t not in sel) and (t in agg.index)]
L335     for t in cand:
L336         within_score = (pd.notna(agg[t]) and agg[t] >= thresh)
L337         within_rank  = (t in ranked_all.index) and (ranked_all.index.get_loc(t) < n_target + keep_buffer)
L338         if within_score or within_rank:
L339             non_inc = [x for x in sel if x not in incumbents]
L340             if not non_inc: break
L341             weakest = min(non_inc, key=lambda x: agg.get(x, -np.inf))
L342             if weakest in sel and agg.get(t, -np.inf) >= agg.get(weakest, -np.inf):
L343                 sel.remove(weakest); sel.append(t)
L344     if len(sel) > n_target:
L345         sel = sorted(sel, key=lambda x: agg.get(x, -1e9), reverse=True)[:n_target]
